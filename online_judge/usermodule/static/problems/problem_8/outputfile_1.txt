#include<bits/stdc++.h>
#define ll long long int
using namespace std;

ll check(string x)
{
    string y = x;
    reverse(y.begin(),y.end());
    return y==x;
}

int main()
{
    string s;
    cin>>s;
    
    ll ans = 3;
    ll len = s.length();
    
    for(ll i=0;i<len/2;i++)
    {
        string x = s.substr(0,i+1);
        string y = s.substr(i+1);
        string w = s.substr(len-i-1);
        string ab = s.substr(i+1,len - 2*(i+1));
        string z = y + x;
        string z1 = w + ab + x;
       // cout<<z<<" "<<s<<endl;
        //cout<<"x:"<<x<<" y:"<<y<<" w:"<<w<<" ab:"<<ab<<" z:"<<z<<" z1:"<<z1<<endl;
        if(check(z) and z!=s)
        {
            ans = min(ans,1ll);
        }
      //  cout<<z1<<endl;
        if(check(z1) and z1!=s)
        {
           // cout<<"heyyy "<<
            ans = min(ans,2ll);
        }
        
    }
    if(ans == 3)
    {
           cout<<"Impossible"<<endl;
    }
    else
    {
        cout<<ans<<endl;
    }
    return 0; 
}
